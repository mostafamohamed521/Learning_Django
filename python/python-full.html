    print(n)  # 0,1,3,4

الدوال — Functions
بلوكات كود قابلة للإعادة — اكتبها مرة واستخدمها في أي مكان.

⚙️
تعريف الدوال بالتفصيل
python
def greet(name):
    return f"مرحباً يا {name}!"

# قيم افتراضية
def power(base, exp=2):
    return base ** exp

power(4)       # 16
power(2, 8)    # 256
power(exp=3, base=5)  # 125

# *args → tuple
def total(*nums):
    return sum(nums)

total(1,2,3)      # 6
total(10,20,30)   # 60

# **kwargs → dict
def info(**data):
    for k, v in data.items():
        print(f"{k}: {v}")

info(name="Ali", age=25)

# إرجاع أكثر من قيمة
def min_max(lst):
    return min(lst), max(lst)

low, high = min_max([3,1,7,2])  # 1, 7

class BankAccount:
    bank_name = "Python Bank"
    total = 0

    def __init__(self, owner, balance=0):
        self.owner = owner
        self._balance = balance
        BankAccount.total += 1

    @property
    def balance(self):
        return self._balance

    @balance.setter
    def balance(self, amount):
        if amount < 0:
            raise ValueError("الرصيد مش ممكن يبقى سالب")
        self._balance = amount

    def deposit(self, amount):
        self._balance += amount

    def withdraw(self, amount):
        if amount > self._balance:
            return False
        self._balance -= amount
        return True

    def __str__(self):
        return f"حساب {self.owner}: {self._balance}"

acc = BankAccount("Ahmed", 1000)
acc.deposit(500)
print(acc)   # حساب Ahmed: 1500

class Shape:
    def __init__(self, color="أبيض"):
        self.color = color
    def area(self):
        raise NotImplementedError

class Circle(Shape):
    def __init__(self, radius, **kwargs):
        super().__init__(**kwargs)
        self.radius = radius
    def area(self):
        return 3.14159 * self.radius**2

class Rectangle(Shape):
    def __init__(self, w, h, **kwargs):
        super().__init__(**kwargs)
        self.w, self.h = w, h
    def area(self):
        return self.w * self.h

shapes = [Circle(5), Rectangle(4,6)]
for s in shapes:
    print(s.area())  # 78.5, 24

Decorators

Decorator = دالة بتاخد دالة وبترجع دالة جديدة بسلوك إضافي.

python
import time
from functools import wraps

def timer(func):
    @wraps(func)
    def wrapper(*args, **kwargs):
        start = time.time()
        result = func(*args, **kwargs)
        print(f"{func.__name__}: {time.time()-start:.3f}s")
        return result
    return wrapper

@timer
def slow():
    time.sleep(1)

slow()

# Decorator بـ arguments
def repeat(times):
    def decorator(func):
        @wraps(func)
        def wrapper(*a, **kw):
            for _ in range(times):
                func(*a, **kw)
        return wrapper
    return decorator

@repeat(3)
def say_hi():
    print("أهلاً!")

say_hi()

def fibonacci(limit):
    a, b = 0, 1
    while a < limit:
        yield a
        a, b = b, a+b
